bool readini::getstring(std::string file)
{
   	std::ifstream infile;
	infile.open(file.data());
	if (!infile.is_open())
		return false;
	else
	{
		int col = 0;
		std::string s;
		std::string title;
		while (getline(infile, s))
		{
			col++;
			int flag = 0;
			std::string key;

			std::string t;
			int is_right = 0;
			int n = s.length();
			for (int i = 0; i < n; i++)

					break;

					flag = 1;
				else if (flag == 2 && s[i] == ']')
					flag = 3;
				else if ((s[i] >= '0'&&s[i] <= '9') || (s[i] >= 'a'&&s[i] <= 'z') || (s[i] >= 'A'&&s[i] <= 'Z') || (s[i] == '"'))

std::string readini::judge_word(const std::string &name, int &p)//相关词法分析
{
	std::string s;
	if (name[p] >= '0'&&name[p] <= '9')
	{
		s=judge_num(name, p);
	}
	if ((name[p] >= 'a'&&name[p] <= 'z')||(name[p] >= 'A'&&name[p] <= 'Z'))
	{
		s=judge_variable(name, p);
	}
	if (name[p] == '"')
	{
		s=judge_string(name, p);
	}
	if (s.length() != 0)
		return s;
	else
		return "";
}
std::string readini::judge_string(const std::string &st,int &p)
{
	std::string ss;
	int i;
	int flag = 0;
	int n = st.length();
	i = p + 1;
	for (; i < n; i++)
	{
		if (st[i] == '"')
		{
			flag = 1;
			p=i;
			return ss;
		}
		ss = ss + st[i];
	}
	return "";
}

}
